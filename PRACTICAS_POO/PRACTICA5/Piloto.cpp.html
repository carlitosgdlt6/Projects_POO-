string Piloto::toCSV () const
{
   std::stringstream aux;

   aux << _nombre << " ; "
       << _nacionalidad << " ; "
       << _numMisiones << " ; "
       << _fechaUltimaMision << " ; "
       << _incidenciasUltimaMision;

   return ( aux.str () );
}

Piloto& Piloto::operator = ( const Piloto& otro )
{
   if ( this != &otro )
   {
      _nombre = otro._nombre;
      _nacionalidad = otro._nacionalidad;
      _numMisiones = otro._numMisiones;
      _fechaUltimaMision = otro._fechaUltimaMision;
      _incidenciasUltimaMision = otro._incidenciasUltimaMision;
   }
   
   return ( *this );
}

//de la clase informe y recplicar los datos de la clase Piloto. Por tanto, tendremos unicamente los atirbutos comunes en ambas clases que son:
//idPiloto, fechaUltimaMision, y un string de datos que contendra: numero indicendias, idNave, idDroide. Es importante, conevrtir estos enteros en
//string para concatenarlos todos en este último atributo
Informe Piloto::generaInforme()const {
   //Antes de nada, vamos a concatenar los atributos que iran en el tercer atributo del metodo. Para ello, teniendo en cuenta de que idNave y idDroide
   //son punteros, tenemos que comprobar que apuntan a algo.
   int idNave=-1;
   int idDroide=-1;

   if (this->_nave!=nullptr) { //Comprobamos que el puntero no sea erroneo
      idNave=this->_nave->getIdSF(); //En caso de que no, sobreescribimos el valor de la id de la nave.
   }
   if (this->_auxiliar!=nullptr) { //Comrpobamos que el puntero no sea erroneo
      idDroide=this->_auxiliar->getIdD(); //En caso de que no lo sea, sobreescribimos el valor de la id del droide
   }

   //Ahora, unimos estos tres atributos en un atributo del metoodo
   std::string datos="Incidencias: "+this->_incidenciasUltimaMision+
                     "Id Nave: "+std::to_string(idNave)+
                        "Id Droide: "+std::to_string(idDroide);
   //Devolvemos ya todo lo pedido en el metodo
   return Informe(this->_idP,this->_fechaUltimaMision,datos);
}
