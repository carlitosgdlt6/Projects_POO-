#include <stdexcept>
#include <sstream>

#include "StarFighter.h"

using std::string;

int StarFighter::_numStarFighters = 0;

StarFighter::StarFighter ( ): StarFighter ( "", "" )
{ }

StarFighter::StarFighter ( string marca, string modelo ): _marca (marca),
                                                          _modelo(modelo),
                                                          _numPlazas(1),
                                                          _parsec(0.0) //Añadimos la inicializacion
{
   _numStarFighters++;
   _idSF = 1000+_numStarFighters;//modificamos esto para que, el id comienze en 1000 y vaya incrementandose, pero el numero de vuelos se mantenga en 0 subiendo de 1 en 1 tambien
}

StarFighter::StarFighter ( const StarFighter& orig ): _marca(orig._marca),
                                                      _modelo(orig._modelo),
                                                      _numPlazas(orig._numPlazas),
                                                      _parsec(orig._parsec)   //Añadimos la inicializacion
{
   _numStarFighters++;
   _idSF = 1000+ _numStarFighters;
}

StarFighter::~StarFighter ( )
{ }

int StarFighter::getNumPlazas ( ) const
{
   return _numPlazas;
}

StarFighter& StarFighter::setModelo ( string modelo )
{
   this->_modelo = modelo;
   return (*this);
}

string StarFighter::getModelo ( ) const
{
   return _modelo;
}

StarFighter& StarFighter::setMarca ( string marca )
{
   this->_marca = marca;
   return (*this);
}

string StarFighter::getMarca ( ) const
{
   return _marca;
}

int StarFighter::getIdSF ( ) const
{
   return _idSF;
}
//Añadimos la implementación de los get y set que nos pedia el apartado
float StarFighter::getParsec()const {
   return _parsec;
}
StarFighter& StarFighter::incrementarParsec(float parsec) {
   if (parsec<0) {
      throw std::invalid_argument("Error, el parsec debe der un valor positivo");
   }
   this->_parsec+=parsec;
   return(*this);
}

string StarFighter::toCSV () const
{
   std::stringstream aux;

   aux << _marca << " ; "
       << _modelo << " ; "
       << _numPlazas<<"; "
      <<_parsec; //AÑADIMOS ESTO

   return ( aux.str () );
}

StarFighter& StarFighter::operator = (const StarFighter& otro)
{
   if ( this != &otro )
   {
      _marca = otro._marca;
      _modelo = otro._modelo;
      _numPlazas = otro._numPlazas;
      _parsec=otro._parsec;
   }
   
   return ( *this );
}
